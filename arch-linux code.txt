код для старых дос dos (не для efi)

code = 1

cfdisk /dev/sda  # Выбирай тип 'dos'
mkfs.ext4 /dev/sda1
mount /dev/sda1 /mnt

code = 2

pacstrap -K /mnt base base-devel linux linux-headers linux-firmware amd-ucode intel-ucode networkmanager git nano bluez bluez-utils
genfstab -U /mnt >> /mnt/etc/fstab
arch-chroot /mnt

код = 3

ln -sf /usr/share/zoneinfo/Europe/Moscow /etc/localtime
hwclock --systohc

# В nano впиши: en_US.UTF-8 UTF-8 и ru_RU.UTF-8 UTF-8
nano /etc/locale.gen
locale-gen
echo "LANG=ru_RU.UTF-8" > /etc/locale.conf

code = 4

useradd -m -G wheel имя_юзера
passwd имя_юзера
pacman -S sudo
# В visudo убери # перед %wheel ALL=(ALL:ALL) ALL
EDITOR=nano visudo

code = 5

pacman -S grub
grub-install --target=i386-pc /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg

code = 6

systemctl enable NetworkManager
systemctl enable systemd-timesyncd
echo "kernel.sysrq = 1" > /etc/sysctl.d/99-sysctl.conf
pacman -S pacman-contrib
systemctl enable paccache.timer
systemctl enable fstrim.timer

code = 7

# Установка всего интерфейса разом
pacman -S xorg-server lxqt sddm openbox pavucontrol-qt qterminal pcmanfm-qt lxqt-archiver lxqt-policykit nm-connection-editor
pacman -S lxappearance-obconf lxqt-qtplugin qt6ct lib32-gtk2 gtk-engine-murrine
pacman -S ttf-liberation ttf-dejavu noto-fonts
systemctl enable sddm

code = 8

pacman -S pipewire pipewire-pulse pipewire-alsa wireplumber
pacman -S ntfs-3g gvfs bash-completion unzip zip p7zip unrar libxkbcommon

code = 9

su - имя_юзера
git clone https://aur.archlinux.org/yay.git
cd yay
makepkg -si
cd ..
rm -rf yay
